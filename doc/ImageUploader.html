def crop
  if model.crop_x.present?
    resize_to_limit(600, 600)
    manipulate! do |img|
      x = model.crop_x.to_i
      y = model.crop_y.to_i
      w = model.crop_w.to_i
      h = model.crop_h.to_i
      img.crop!(x, y, w, h)
    end
  end
end

def extension_white_list
  if model.image_type == "Avatar"
    %w(jpg jpeg gif png)
  elsif model.image_type == "Ingredient"
    %w(pdf doc docx xls xlsx)
  elsif model.image_type == "Music"
    %w(mp3 wav wma ogg)
  end
end
